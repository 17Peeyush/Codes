class RandomizedSet {
public:
    vector<int>idx;
    unordered_map<int,int>mp;
    RandomizedSet() {
        
    }
    
    bool insert(int val) {
        if(mp.find(val)!=mp.end())
            return false;
        idx.push_back(val);
        int n=idx.size();
        mp[val]=n-1;
        return true;
    }
    
    bool remove(int val) {
        if(mp.find(val)==mp.end())
            return false;
        int n=idx.size(),cdx=mp[val];
        idx[cdx]=idx[n-1];
        mp[idx[cdx]]=cdx;
        idx.pop_back();
        mp.erase(val); 
        return true;
    }
    
    int getRandom() {
        return idx[rand()%idx.size()];
    }
};
