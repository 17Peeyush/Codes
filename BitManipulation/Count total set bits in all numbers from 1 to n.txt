Problem:https://www.geeksforgeeks.org/count-total-set-bits-in-all-numbers-from-1-to-n/
Time Complexity: O(logn)
Auxillary Space: O(1)
Code:
int getpow(int n)
    {
        int count=0,num=1;
        while(num<=n)
        {
            count++;
            num=1<<count;
        }
        return count -1;
    }
    int countSetBits(int n)
    {
        if(n<=1)
        return n;
        int x= getpow(n);
        return (pow(2,x-1)*x)+(n-pow(2,x)+1)+countSetBits(n-pow(2,x));
    }
