//Function to find the data of nth node from the end of a linked list.
int getNthFromLast(Node *head, int n)
{
       // Your code here
       
    struct Node *f=head;
    struct Node *p=head;
    n--;
    while(n--&&f!=nullptr)
    {
        f=f->next;
    }
    if(f==nullptr)
    return -1;
    while(f->next!=nullptr)
    {
        p=p->next;
        f=f->next;
    }
    return p->data;
}
