Approach:(Greedy)
Suppose there are m contest of importance 1.
k Contest won of importance 1 with minimum luck.
All contest lost of importance 0 and m-k contest of importance 1.

Code:
int luckBalance(int k, vector<vector<int>> contests) {
    int luck=0,n=contests.size();
    if(k!=0)
    {
    priority_queue<int>q;
    for(int i=0;i<n;i++)
    {
        if(contests[i][1]==1)
        {
            if(!k)
            {
                int top=-q.top();
                if(contests[i][0]<top)
                {
                    luck-=contests[i][0];
                }
                else
                {
                    luck+=contests[i][0];
                    luck-=2*top;
                    q.pop();
                    q.push(-contests[i][0]);
                }
            }
            else
            {
                luck+=contests[i][0];
                q.push(-contests[i][0]);
                k--;
            }
            
        }
        else
        {
            luck+=contests[i][0];
        }
    }
    }
    else
    {
        for(int i=0;i<n;i++)
        {
            if(contests[i][1]==0)
                luck+=contests[i][0];
            else
                luck-=contests[i][0];
        }
    }
    return luck;
}
