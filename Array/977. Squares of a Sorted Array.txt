Problem:https://leetcode.com/problems/squares-of-a-sorted-array/description
Time Complexity: O(nlogn)
Auxillary Space: O(nlogn)
func sortedSquares(nums []int) []int {
    size := len(nums)
    res := make([] int, size)
    for idx, val := range nums{
        res[idx]=val*val
    }
    sort.Ints(res)
    return res
}
